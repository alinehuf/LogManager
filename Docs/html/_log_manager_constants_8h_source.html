82/* Enum for the different events: don't change the enum name !
83 * The values of the enum are not important, but the name and type of the enum must not be changed,
84 * as they are used in the Blueprint Function Library.
85 *
86 * EEvent identifies the type of a log entry written to a JSON log file.
87 * Each value corresponds to a specific stage or action in an XP (experiment) session.
88 */
89UENUM(BlueprintType)
90enum class EEvent : uint8 {
91 // For the start survey or an XP
92 START_SURVEY,
93 // Start of XP stage: records user id, xp settings...
94 XP_START,
95 // Records a specific action of the user
96 BOX_EVENT,
97 // Records the interaction of the user with a cube
98 CUBE_INTERACTION,
99 // End of XP stage: allow you to check is log file is properly completed
100 XP_STOP,
101 // For the final survey or each XP run
102 FINAL_SURVEY
103};
